📜 Referência rápida — ALTER TABLE
SQL
-- Adicionar coluna a tabela existente
ALTER TABLE produto
  ADD COLUMN descricao VARCHAR(255);

-- Adicionar FK a tabela existente
ALTER TABLE produto
  ADD COLUMN id_categoria INT,
  ADD FOREIGN KEY (id_categoria)
    REFERENCES categoria(id_categoria);
